// blocks/portfolio-grid/index.js

import { __ }                              from '@wordpress/i18n';
import { useBlockProps, InspectorControls } from '@wordpress/block-editor';
import {
    PanelBody,
    RangeControl,
    SelectControl,
    ToggleControl,
    Spinner,
    Placeholder,
}                                          from '@wordpress/components';
import { useSelect }                       from '@wordpress/data';
import { registerBlockType }               from '@wordpress/blocks';

import metadata from './block.json';

function Edit( { attributes, setAttributes } ) {
    const { columns, perPage, category, showExcerpt, orderby } = attributes;
    const blockProps = useBlockProps( {
        className: `portfolio-grid portfolio-grid--cols-${ columns }`,
    } );

    // Cargar categorías para el selector
    const categories = useSelect( ( select ) => {
        return select( 'core' ).getEntityRecords( 'taxonomy', 'portfolio-category', {
            per_page: 100,
            hide_empty: true,
        } );
    }, [] );

    // Construir las opciones del selector de categoría
    const categoryOptions = [
        { label: __( 'Todas las categorías', 'ninja-portfolio' ), value: '' },
        ...( categories ?? [] ).map( ( cat ) => ( {
            label: cat.name,
            value: cat.slug,
        } ) ),
    ];

    // Cargar proyectos con los filtros actuales para la preview
    const { posts, isLoading } = useSelect(
        ( select ) => {
            const query = {
                per_page:           perPage,
                post_status:        'publish',
                orderby,
                order:              'desc',
                portfolio_category: category || undefined,
            };

            return {
                posts:     select( 'core' ).getEntityRecords( 'postType', 'portfolio', query ),
                isLoading: select( 'core' ).isResolving( 'getEntityRecords', [
                    'postType',
                    'portfolio',
                    query,
                ] ),
            };
        },
        [ perPage, category, orderby ]
    );

    return (
        <>
            <InspectorControls>
                <PanelBody title={ __( 'Diseño', 'ninja-portfolio' ) }>
                    <RangeControl
                        label={ __( 'Columnas', 'ninja-portfolio' ) }
                        value={ columns }
                        onChange={ ( value ) => setAttributes( { columns: value } ) }
                        min={ 1 }
                        max={ 4 }
                    />
                    <RangeControl
                        label={ __( 'Proyectos a mostrar', 'ninja-portfolio' ) }
                        value={ perPage }
                        onChange={ ( value ) => setAttributes( { perPage: value } ) }
                        min={ 1 }
                        max={ 24 }
                    />
                    <ToggleControl
                        label={ __( 'Mostrar extracto', 'ninja-portfolio' ) }
                        checked={ showExcerpt }
                        onChange={ ( value ) => setAttributes( { showExcerpt: value } ) }
                    />
                </PanelBody>
                <PanelBody title={ __( 'Filtrado', 'ninja-portfolio' ) } initialOpen={ false }>
                    <SelectControl
                        label={ __( 'Categoría', 'ninja-portfolio' ) }
                        value={ category }
                        options={ categoryOptions }
                        onChange={ ( value ) => setAttributes( { category: value } ) }
                    />
                    <SelectControl
                        label={ __( 'Ordenar por', 'ninja-portfolio' ) }
                        value={ orderby }
                        options={ [
                            { label: __( 'Fecha', 'ninja-portfolio' ),      value: 'date' },
                            { label: __( 'Título', 'ninja-portfolio' ),     value: 'title' },
                            { label: __( 'Orden manual', 'ninja-portfolio' ), value: 'menu_order' },
                        ] }
                        onChange={ ( value ) => setAttributes( { orderby: value } ) }
                    />
                </PanelBody>
            </InspectorControls>

            <div { ...blockProps }>
                { isLoading && (
                    <Placeholder label={ __( 'Portfolio Grid', 'ninja-portfolio' ) }>
                        <Spinner />
                    </Placeholder>
                ) }

                { ! isLoading && ( ! posts || posts.length === 0 ) && (
                    <Placeholder label={ __( 'Portfolio Grid', 'ninja-portfolio' ) }>
                        <p>{ __( 'No hay proyectos publicados.', 'ninja-portfolio' ) }</p>
                    </Placeholder>
                ) }

                { ! isLoading && posts && posts.length > 0 && (
                    <div className="portfolio-grid__inner">
                        { posts.map( ( post ) => (
                            <article key={ post.id } className="portfolio-card portfolio-card--editor">
                                { post._embedded?.[ 'wp:featuredmedia' ]?.[ 0 ]?.source_url && (
                                    <div className="portfolio-card__thumb">
                                        <img
                                            src={ post._embedded[ 'wp:featuredmedia' ][ 0 ].source_url }
                                            alt={ post.title.rendered }
                                            loading="lazy"
                                        />
                                    </div>
                                ) }
                                <div className="portfolio-card__body">
                                    <h3 className="portfolio-card__title">
                                        { post.title.rendered }
                                    </h3>
                                    { showExcerpt && post.excerpt?.rendered && (
                                        <div
                                            className="portfolio-card__excerpt"
                                            dangerouslySetInnerHTML={ {
                                                __html: post.excerpt.rendered,
                                            } }
                                        />
                                    ) }
                                </div>
                            </article>
                        ) ) }
                    </div>
                ) }
            </div>
        </>
    );
}

function Save() {
    return null;
}

registerBlockType( metadata.name, { edit: Edit, save: Save } );
