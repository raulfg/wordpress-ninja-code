import { __ } from '@wordpress/i18n';
import { useBlockProps, InspectorControls } from '@wordpress/block-editor';
import { PanelBody, RangeControl, SelectControl } from '@wordpress/components';

export function Edit( { attributes, setAttributes } ) {
    const { columns, category } = attributes;
    const blockProps = useBlockProps( {
        style: { '--columns': columns },
    } );

    return (
        <>
            <InspectorControls>
                <PanelBody title={ __( 'Settings', 'ninjatheme' ) }>
                    <RangeControl
                        label={ __( 'Columns', 'ninjatheme' ) }
                        value={ columns }
                        onChange={ ( value ) => setAttributes( { columns: value } ) }
                        min={ 1 }
                        max={ 6 }
                    />
                    <SelectControl
                        label={ __( 'Category', 'ninjatheme' ) }
                        value={ category }
                        options={ [
                            { label: __( 'All', 'ninjatheme' ), value: '' },
                            { label: 'Web', value: 'web' },
                            { label: 'Branding', value: 'branding' },
                            { label: 'Photography', value: 'fotografia' },
                        ] }
                        onChange={ ( value ) => setAttributes( { category: value } ) }
                    />
                </PanelBody>
            </InspectorControls>
            <div { ...blockProps }>
                <p className="portfolio-grid__placeholder">
                    { __( 'Portfolio Grid — frontend preview', 'ninjatheme' ) }
                </p>
            </div>
        </>
    );
}

export function Save() {
    return null;
}
